# show_timeline.py
import sys
from pathlib import Path

path = Path("evidence_index") / "timeline.jsonl"

n = int(sys.argv[1]) if len(sys.argv) > 1 else 10

if not path.exists():
    print("No timeline yet.")
    raise SystemExit

lines = path.read_text(encoding="utf-8").splitlines()
for line in lines[-n:]:
    print(line)
